73      /**
74       * Http error response from the error bean.
75       * 
76       * @param status
77       * @param detailedMsg
78       * @param request
79       * @param response
80       * @param updateCountsFor
81       * @throws IOException
82       */
83  	protected final void errorResponse(Status status, String detailedMsg,
84  			HttpServletRequest request, HttpServletResponse response, Set<LogEvent> updateCountsFor) 
85  					throws IOException {
86  		
87  		if(updateCountsFor == null)
88  			updateCountsFor = new HashSet<LogEvent>();
89  		
90  		// to count the error (code), also add -
91  		updateCountsFor.add(LogEvent.from(status));
92  		
93  		//problems with logging subsystem should not fail the entire service
94  		try {
95  			service.log(updateCountsFor, clientIpAddress(request));
96  		} catch (Throwable ex) {
97  			log.error("LogUtils.log failed", ex);
98  		}
99  		
100 		response.sendError(status.getErrorCode(), 
101 			status.getErrorMsg() + "; " + detailedMsg);
102 	}

105 	/**
106 	 * Builds an error message from  
107 	 * the web parameters binding result
108 	 * if there're errors.
109 	 * 
110 	 * @param bindingResult
111 	 * @return
112 	 */
113 	protected final String errorFromBindingResult(BindingResult bindingResult) {
114 		StringBuilder sb = new StringBuilder();
115 		for (FieldError fe : bindingResult.getFieldErrors()) {
116 			Object rejectedVal = fe.getRejectedValue();
117 			if(rejectedVal instanceof Object[]) {
118 				if(((Object[]) rejectedVal).length > 0) {
119 					rejectedVal = Arrays.toString((Object[])rejectedVal);
120 				} else {
121 					rejectedVal = "empty array";
122 				}
123 			}
124 			sb.append(fe.getField() + " was '" + rejectedVal + "'; "
125 					+ fe.getDefaultMessage() + ". ");
126 		}
127 		
128 		return sb.toString();
129 	}

132 	/**
133 	 * Writes the query results to the HTTP response
134 	 * output stream.
135 	 * 
136 	 * @param resp
137 	 * @param writer
138 	 * @param request
139 	 * @param response
140 	 * @param updateCountsFor
141 	 * @throws IOException
142 	 */
143 	protected final void stringResponse(ServiceResponse resp, 
144 			Writer writer, HttpServletRequest request, 
145 			HttpServletResponse response, Set<LogEvent> updateCountsFor) throws IOException 
146 	{
147 		if(resp instanceof ErrorResponse) {
148 			
149 			errorResponse(((ErrorResponse) resp).getStatus(), 
150 					((ErrorResponse) resp).toString(), request, response, updateCountsFor);
151 			
152 		} 
153 		else if(resp.isEmpty()) {
154 			log.warn("stringResponse: I got an empty ServiceResponce " +
155 				"(must be already converted to the ErrorResponse)");
156 			
157 			errorResponse(NO_RESULTS_FOUND, "no results found", 
158 					request, response, updateCountsFor);
159 			
160 		} 
161 		else {
162 			response.setContentType("text/plain");
163 			DataResponse dresp = (DataResponse) resp;
164 
165 			log.debug("QUERY RETURNED " + dresp.getData().toString().length() + " chars");
166 			
167 			// take care to count provider's data accessed events
168 			Set<String> providers = dresp.getProviders();
169 			updateCountsFor.addAll(LogEvent.fromProviders(providers));
170 			
171 			//log to the db (for analysis and reporting)
172 			//problems with logging subsystem should not fail the entire service
173 			try {
174 				service.log(updateCountsFor, clientIpAddress(request));
175 			} catch (Throwable ex) {
176 				log.error("LogUtils.log failed", ex);
177 			}
178 			
179 			writer.write(dresp.getData().toString());
180 		}
181 	}

184 	/**
185 	 * Resizes the image.
186 	 * 
187 	 * @param img
188 	 * @param width
189 	 * @param height
190 	 * @param background
191 	 * @return
192 	 */
193 	public final BufferedImage scaleImage(BufferedImage img, int width, int height,
194 	        Color background) {
195 	    int imgWidth = img.getWidth();
196 	    int imgHeight = img.getHeight();
197 	    if (imgWidth*height < imgHeight*width) {
198 	        width = imgWidth*height/imgHeight;
199 	    } else {
200 	        height = imgHeight*width/imgWidth;
201 	    }
202 	    BufferedImage newImage = new BufferedImage(width, height,
203 	            BufferedImage.TYPE_INT_RGB);
204 	    Graphics2D g = newImage.createGraphics();
205 	    try {
206 	        g.setRenderingHint(RenderingHints.KEY_INTERPOLATION,
207 	                RenderingHints.VALUE_INTERPOLATION_BICUBIC);
208 	        if(background != null)
209 	        	g.setBackground(background);
210 	        g.clearRect(0, 0, width, height);
211 	        g.drawImage(img, 0, 0, width, height, null);
212 	    } finally {
213 	        g.dispose();
214 	    }
215 	    return newImage;
216 	}

219 	/**
220 	 * Extracts the client's IP from the request headers.
221 	 * 
222 	 * @param request
223 	 * @return
224 	 */
225 	public static final String clientIpAddress(HttpServletRequest request) {
226 		
227 		String ip = request.getHeader("X-Forwarded-For");		
228 		if (ip == null || ip.isEmpty() || "unknown".equalsIgnoreCase(ip)) {  
229             ip = request.getHeader("Proxy-Client-IP");  
230         }  
231         if (ip == null || ip.isEmpty() || "unknown".equalsIgnoreCase(ip)) {  
232             ip = request.getHeader("WL-Proxy-Client-IP");  
233         }  
234         if (ip == null || ip.isEmpty() || "unknown".equalsIgnoreCase(ip)) {  
235             ip = request.getHeader("HTTP_CLIENT_IP");  
236         }  
237         if (ip == null || ip.isEmpty() || "unknown".equalsIgnoreCase(ip)) {  
238             ip = request.getHeader("HTTP_X_FORWARDED_FOR");  
239         }  
240         if (ip == null || ip.isEmpty() || "unknown".equalsIgnoreCase(ip)) {  
241             ip = request.getRemoteAddr();  
242         }  
243 		
244         return ip;
245 	}
