51  
52  	static OntologyManagerCvRepository cvRepository;	
53  	
54  	static {
55  		final Properties cfg = new Properties();
56  		cfg.put("SO", "classpath:so.obo");
57  		cfg.put("MI", "classpath:mi.obo");
58  		cfg.put("MOD", "classpath:mod.obo");
59  		cfg.put("GO", "classpath:go.obo");
60  		cvRepository = new OntologyManagerCvRepository(cfg);
61  	}
62  	
63  	@Test
64  	public void ontologyLoading() {
65  		Collection<String> ontologyIDs = cvRepository.getOntologyIDs();
66  		assertTrue(ontologyIDs.contains("GO"));
67  		assertEquals("gene ontology", cvRepository.getOntology("GO").getName().toLowerCase());
68  		assertTrue(ontologyIDs.contains("SO"));
69  		assertTrue(ontologyIDs.contains("MI"));
70  		assertTrue(ontologyIDs.contains("MOD"));
71  	}
72  
73  	/**
74  	 * Test method for
75  	 * {@link cpath.dao.OntologyManagerCvRepository#ontologyTermsToUris(java.util.Collection)}
76  	 */
77  	@Test
78  	public final void testOntologyTermsToUris() {
79  		OntologyTermI term = new OntologyTermImpl("GO", "GO:0005654", "nucleoplasm");
80  		Set<OntologyTermI> terms = new HashSet<OntologyTermI>();
81  		terms.add(term);
82  		Set<String> urns = cvRepository.ontologyTermsToUris(terms);
83  		assertFalse(urns.isEmpty());
84  		assertTrue(urns.size() == 1);
85  		String urn = urns.iterator().next();
86  		assertEquals("http://identifiers.org/go/GO:0005654", urn);
87  	}

94  	public final void testSearchForTermByAccession() {
95  		OntologyTermI term = cvRepository.findTermByAccession("GO:0005654");

97  		assertEquals("nucleoplasm", term.getPreferredName());
98  	}

141 	public final void testGetObject() {
142 		CellularLocationVocabulary cv = cvRepository.getControlledVocabulary(
143 				"urn:miriam:obo.go:GO%3A0005737",CellularLocationVocabulary.class);
144 		assertNotNull(cv);
145 		cv = null;
146 		cv = cvRepository.getControlledVocabulary( //using now deprecated URL
147 				"http://identifiers.org/obo.go/GO:0005737",CellularLocationVocabulary.class);
148 		assertNotNull(cv);
149 		//same
150 		cv = cvRepository.getControlledVocabulary(
151 				"http://identifiers.org/go/GO:0005737",CellularLocationVocabulary.class);
152 		assertNotNull(cv);
153 	}
