001package cpath.jpa;
002
003
004import javax.persistence.AttributeOverride;
005import javax.persistence.AttributeOverrides;
006import javax.persistence.Column;
007import javax.persistence.Embedded;
008import javax.persistence.Entity;
009import javax.persistence.Table;
010import javax.persistence.UniqueConstraint;
011
012import org.hibernate.annotations.DynamicInsert;
013import org.hibernate.annotations.DynamicUpdate;
014import org.springframework.util.Assert;
015
016
017
018/**
019 * @author rodche
020 *
021 */
022@Entity
023@DynamicUpdate
024@DynamicInsert
025//uniqueConstraints: 'type' is not listed because 'name' is enough (should never use same log name with different types)
026@Table(uniqueConstraints = @UniqueConstraint(columnNames = {"date", "name", "addr"}))
027public class LogEntity extends AbstractEntity {
028                
029        @Embedded
030        @AttributeOverrides({
031            @AttributeOverride(name="type", column=@Column(name="type")),
032            @AttributeOverride(name="name", column=@Column(name="name"))
033        })
034        private LogEvent event;
035        
036        @Column
037        private String addr;
038        
039        @Embedded
040        @AttributeOverrides({
041            @AttributeOverride(name="country", column=@Column(name="country")),
042            @AttributeOverride(name="region", column=@Column(name="region")),
043            @AttributeOverride(name="city", column=@Column(name="city"))
044        })
045        private Geoloc geoloc;
046
047        @Column(nullable=false)
048        private String date;
049        
050        @Column(nullable=false)
051        private Long count;     
052        
053        public LogEntity() {
054        }
055        
056        /**
057         * 
058         * @param date ISO date (yyyy-MM-dd)
059         * @param event
060         * @param ipAddress
061         */
062        public LogEntity(String date, LogEvent event, String ipAddress) {
063                Assert.notNull(event);
064                Assert.notNull(event.getType());
065                Assert.notNull(event.getName());
066                Assert.notNull(date);
067                Assert.notNull(ipAddress);
068                
069                setEvent(event);
070                setCount(0L);
071                setDate(date);
072                        
073                setAddr(ipAddress);
074                geoloc = Geoloc.fromIpAddress(ipAddress);
075                setGeoloc(geoloc);
076        }
077        
078        
079        public String getAddr() {
080                return addr;
081        }
082        public void setAddr(String addr) {
083                this.addr = addr;
084        }       
085        
086        public Long getCount() {
087                return count;
088        }
089        public void setCount(Long count) {
090                this.count = count;
091        }
092
093        public String getDate() {
094                return date;
095        }
096        public void setDate(String date) {
097                this.date = date;
098        }
099        
100        public Geoloc getGeoloc() {
101                return geoloc;
102        }
103        public void setGeoloc(Geoloc geoloc) {
104                this.geoloc = geoloc;
105        }
106
107        public LogEvent getEvent() {
108                return event;
109        }
110        public void setEvent(LogEvent event) {
111                this.event = event;
112        }
113        
114}
