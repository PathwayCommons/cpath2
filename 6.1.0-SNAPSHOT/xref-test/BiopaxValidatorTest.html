123     @Test
124 	public void testProteinModificationFeatureCvRule() {
125     	CvRule rule = (CvRule) context.getBean("proteinModificationFeatureCvRule");
126     	
127     	//System.out.print("proteinModificationFeatureCvRule valid terms are: " + rule.getValidTerms().toString());
128     	assertTrue(rule.getValidTerms().contains("(2S,3R)-3-hydroxyaspartic acid".toLowerCase()));
129     	
130     	SequenceModificationVocabulary cv = level3.create(SequenceModificationVocabulary.class, "MOD_00036");
131     	cv.addTerm("(2S,3R)-3-hydroxyaspartic acid");
132     	ModificationFeature mf = level3.create(ModificationFeature.class, "MF_MOD_00036");
133     	mf.setModificationType(cv);
134     	Validation v = new Validation(new IdentifierImpl(), "", true, null, 0, null); // auto-fix=true - fixex "no xref" error
135    		rule.check(v, mf); 
136    		
137    		assertEquals(0, v.countErrors(mf.getRDFId(), null, "illegal.cv.term", null, false, false));
138    		assertEquals(1, v.countErrors(mf.getRDFId(), null, "no.xref.cv.terms", null, false, false)); //- one but fixed though -
139    		assertEquals(0, v.countErrors(null, null, null, null, false, true)); //- no unfixed errors
140 	}

143 	@Test
144 	public final void testNormalizeTestFile() throws IOException {
145 		SimpleIOHandler simpleReader = new SimpleIOHandler();
146 		simpleReader.mergeDuplicates(true);
147 		
148 		Normalizer normalizer = new Normalizer();
149 		String base = "test/";
150 		normalizer.setXmlBase(base);
151 		
152 		Model m = simpleReader.convertFromOWL(getClass().getResourceAsStream("/biopax-level3-test.owl"));
153 		normalizer.normalize(m);
154 
155 		/* Following assertions were changed since using the biopax-validator v3 (2012/11/14).
156 		 * Normalizer there does not turn DB or ID values to upper case when generating a new xref URI anymore..." (that was actually a bad idea)
157 		 * "c00022", by the way, is technically an illegal KEGG identifier (must be C00022), - 
158 		 * and it won't pass our Premerger (import pipeline) stage without the critical error being reported.
159 		 * Unfortunately, the normalizer alone cannot always fix such issues (it just generates consistent xref URIs), 
160 		 * because there are less trivial cases than where one could simply convert the first symbol to upper case...;
161 		 * and, more important, xref.id value capitalization does usually matter...
162 		 */
163 		assertTrue(m.containsID(Normalizer.uri(base, "KEGG COMPOUND", "c00022", UnificationXref.class)));
164 		assertTrue(m.containsID(Normalizer.uri(base, "KEGG COMPOUND", "C00022", UnificationXref.class)));
165 		m = null;
166 		
167 		// However, using the validator (with autofix=true) and then - normalizer (as it's done in Premerger) together
168 		// will, in fact, fix and merge these two xrefs
169 		m = simpleReader.convertFromOWL(getClass().getResourceAsStream("/biopax-level3-test.owl"));
170 		Validation v = new Validation(new IdentifierImpl(), null, true, null, 0, null);
171 		v.setModel(m);
172 		validator.validate(v);
173 		validator.getResults().remove(v);
174 		m = (Model)v.getModel();
175 		normalizer.normalize(m);
176 
177 		assertFalse(m.containsID(Normalizer.uri(base, "KEGG COMPOUND", "c00022", UnificationXref.class)));
178 		assertTrue(m.containsID(Normalizer.uri(base, "KEGG COMPOUND", "C00022", UnificationXref.class)));
179 	}
