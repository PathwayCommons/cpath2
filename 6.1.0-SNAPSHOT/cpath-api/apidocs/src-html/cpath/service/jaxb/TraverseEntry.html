001package cpath.service.jaxb;
002
003import java.io.Serializable;
004import java.util.ArrayList;
005import java.util.List;
006import javax.xml.bind.annotation.XmlAccessType;
007import javax.xml.bind.annotation.XmlAccessorType;
008import javax.xml.bind.annotation.XmlAttribute;
009import javax.xml.bind.annotation.XmlTransient;
010import javax.xml.bind.annotation.XmlType;
011
012
013@XmlAccessorType(XmlAccessType.FIELD)
014@XmlType(name = "TraverseEntry")
015public class TraverseEntry implements Serializable {
016    @XmlAttribute(required = true)
017    private String uri;
018    
019    private List<String> value;
020
021    public TraverseEntry() {
022        }
023
024    public String getUri() {
025        return uri;
026    }
027    public void setUri(String value) {
028        this.uri = value;
029    }
030        
031        public List<String> getValue() {
032                if(value == null) {
033                        value = new ArrayList<String>();
034                }
035                return value;
036        }
037        public void setValue(List<String> value) {
038                this.value = value;
039        }
040
041        @Override
042        public String toString() {
043                return getValue().toString();
044        }
045        
046        //package-private
047        @XmlTransient
048        boolean isEmpty() {
049                return getValue().isEmpty();
050        }
051}
