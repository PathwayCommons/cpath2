001package cpath.jpa;
002
003import javax.persistence.GeneratedValue;
004import javax.persistence.GenerationType;
005import javax.persistence.Id;
006import javax.persistence.MappedSuperclass;
007
008
009@MappedSuperclass
010public class AbstractEntity {
011
012        @Id
013        @GeneratedValue(strategy = GenerationType.AUTO)
014        private Long id;
015
016        /**
017         * Returns the identifier of the entity.
018         * 
019         * @return the id
020         */
021        public Long getId() {
022                return id;
023        }
024
025        
026        /* 
027         * (non-Javadoc)
028         * @see java.lang.Object#equals(java.lang.Object)
029         */
030        @Override
031        public boolean equals(Object obj) {
032
033                if (this == obj) {
034                        return true;
035                }
036
037                if (this.id == null || obj == null || !(this.getClass().equals(obj.getClass()))) {
038                        return false;
039                }
040
041                AbstractEntity that = (AbstractEntity) obj;
042
043                return this.id.equals(that.getId());
044        }
045
046        /* 
047         * (non-Javadoc)
048         * @see java.lang.Object#hashCode()
049         */
050        @Override
051        public int hashCode() {
052                return id == null ? 0 : id.hashCode();
053        }
054}
