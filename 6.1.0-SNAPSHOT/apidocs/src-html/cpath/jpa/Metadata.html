053        // data types
054        PSI_MI(true), // interactions to be converted to BioPAX L3 format
055        PSI_MITAB(true), // interactions to be converted to PSI-MI then to BioPAX L3 format
056                BIOPAX(true), // pathways and interactions in BioPAX L2 or L3 format
057                WAREHOUSE(false), // warehouse data to be converted to BioPAX and used during the merge stage
058                MAPPING(false); //extra gene/protein id-mapping data (two column, TSV format: "some id or name" \t "primary uniprot/chebi AC")

345        /**
346         * Creates a new Provenance from this Metadata and sets
347         * if to all Entity class objects in the model. 
348         * 
349         * Removes all other Provenance instances and 
350         * corresponding dataSource property values
351         * from the model.
352         * 
353         * @param model BioPAX model to update
354         */
355        public void setProvenanceFor(Model model) {
356                Provenance pro = null;
357                
358                // we create URI from the Metadata identifier and version.
359                final String uri = getUri();
360                pro = (model.containsID(uri)) 
361                        ? (Provenance) model.getByID(uri)
362                        : model.addNew(Provenance.class, uri);
363                
364                // parse/set names
365                String displayName = getName().iterator().next();
366                pro.setDisplayName(displayName);
367                pro.setStandardName(standardName());
368                
369                if(getName().size() > 2)
370                        for(int i=2; i < getName().size(); i++)
371                                pro.addName(getName().get(i));
372                
373                // add additional info about the current version, source, identifier, etc...
374                final String loc = getUrlToData(); 
375                pro.addComment("Source " + 
376                        //skip for a local or empty (default) location
377                        ((loc.startsWith("http:") || loc.startsWith("ftp:")) ? loc : "") 
378                        + " type: " + getType() + ", " + getDescription());
379                
380                // replace for all entities
381                for (org.biopax.paxtools.model.level3.Entity ent : model.getObjects(org.biopax.paxtools.model.level3.Entity.class)) {
382                        for(Provenance ds : new HashSet<Provenance>(ent.getDataSource()))
383                                ent.removeDataSource(ds);
384                        ent.addDataSource(pro);
385                }
386                
387                // remove dangling Provenance from the model
388                ModelUtils.removeObjectsIfDangling(model, Provenance.class);
389        }

391        /**
392         * Returns the standard name (the second one in the name list),
393         * if present, otherwise - returns the first name (display name)
394         * 
395         * @return
396         */
397        public String standardName() {
398                //also capitalize (can be extremely useful...)
399                if(name.size() > 1)
400                        return StringUtils.capitalize(name.get(1));
401                else 
402                        return StringUtils.capitalize(name.get(0));
403        }
