001package cpath.service.jaxb;
002
003import java.util.Set;
004
005import javax.xml.bind.annotation.*;
006
007/**
008 * An internal service bean, any-data response type.
009 * 
010 * This is normally not for XML marshalling/unmarshalling, 
011 * but rather for exchanging data between DB, service, and web tiers.
012 *
013 */
014@XmlTransient
015public class DataResponse extends ServiceResponse {
016
017        private Object data; // BioPAX OWL, String, List, or any other data
018        @XmlTransient
019        private Set<String> providers; //pathway data provider standard names (for logging/stats)
020        
021        public DataResponse() {
022        }
023        
024        public Object getData() {
025                return data;
026        }
027        public void setData(Object data) {
028                this.data = data;
029        }
030        
031        @Override
032        @XmlTransient
033        public boolean isEmpty() {
034                return  (data == null || data.toString().trim().isEmpty());
035        }
036        
037        @XmlTransient
038        public Set<String> getProviders() {
039                return providers;
040        }
041        public void setProviders(Set<String> providers) {
042                this.providers = providers;
043        }
044}
