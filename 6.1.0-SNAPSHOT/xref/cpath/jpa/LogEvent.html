21  /**
22   * A pair of type and name that describes log entry.
23   * 
24   * A convenience bean for logging
25   * a server access event from different 
26   * perspectives. A list of such value 
27   * pairs can be associated with a request IP address
28   * and date and used to increment corresponding access counts 
29   * all at once, e.g., to simultaneously log which data sources
30   * were used, format, command, filename, etc., from one web request.
31   * 
32   * @see LogEntity
33   * 
34   * @author rodche
35   *
36   */
37  @Embeddable
38  public class LogEvent {
39  	
40  	@Column(nullable=false)
41  	@Enumerated(EnumType.STRING)
42  	private LogType type;
43  
44  	@Column(nullable=false)
45  	private String name;
46  	
47  	/**
48  	 * a special constant event type, because there is no DOWNLOAD command in the cpath2 api.
49  	 */
50  	public static LogEvent DOWNLOAD = new LogEvent(LogType.COMMAND, "DOWNLOAD");
51  	
52  	/**
53  	 * a special constant event type, because there is no IDMAPPING command in the cpath2 api.
54  	 */
55  	public static LogEvent IDMAPPING = new LogEvent(LogType.COMMAND, "IDMAPPING");
56  	
57  	/**
58  	 * a special constant event type for total access counts
59  	 */
60  	public static LogEvent TOTAL = new LogEvent(LogType.TOTAL, LogType.TOTAL.description);
61  	
62  	/**
63  	 * a special constant event type for other formats access counts (e.g., XML, JSON search, id-mapping query results)
64  	 */
65  	public static LogEvent FORMAT_OTHER = new LogEvent(LogType.FORMAT, "OTHER");
66  	
67  	
68  	public LogEvent() {

72  		Assert.notNull(type);

106 	public static Set<LogEvent> fromProviders(Collection<String> providers) {
107 		Set<LogEvent> set = new HashSet<LogEvent>();
108 		for(String prov : providers)
109 			set.add(new LogEvent(LogType.PROVIDER, prov));
110 		return set;
111 	}

114 	/**
115 	 * Creates a list of things to update counts for -
116 	 * name, format, provider - from the 
117 	 * auto-generated data archive file.
118 	 * 
119 	 * @param filename see {@link CPathSettings#biopaxExportFileName(String)}
120 	 * 			for how it's created.
121 	 * @return
122 	 */
123 	public static Set<LogEvent> fromDownloads(String filename) {
124 		Set<LogEvent> set = new HashSet<LogEvent>();
125 
126 		set.add(new LogEvent(LogType.FILE, filename));
127 		set.add(LogEvent.DOWNLOAD);
128 		
129 		// extract the orig. data source's standard name -
130 		// first, remove common prefix (incl. cPath2 instance name and ver.)
131 		if(filename.startsWith(CPathSettings.getInstance().exportArchivePrefix())) {
132 			int idx = CPathSettings.getInstance().exportArchivePrefix().length();
133 			String s = filename.substring(idx);
134 			String[] parts = s.split("\\.");
135 			//a hack: in order to skip for by-organism and special archives
136 			if(Character.isUpperCase(parts[0].charAt(0)) 
137 					&& !"All".equalsIgnoreCase(parts[0])
138 					&& !"Warehouse".equalsIgnoreCase(parts[0])
139 					&& !"Detailed".equalsIgnoreCase(parts[0])) {
140 				set.add(new LogEvent(LogType.PROVIDER, parts[0]));
141 			}
142 
143 			// extract the format
144 			OutputFormat format = null; 
145 			try { 
146 				format = OutputFormat.valueOf(parts[1]);
147 			} catch(Exception e) {
148 
149 			}
150 			if(format != null)
151 				set.add(LogEvent.from(format));
152 			else {
153 				set.add(LogEvent.FORMAT_OTHER);
154 			}
155 		} else {
156 			set.add(LogEvent.FORMAT_OTHER);
157 		}
158 			
159 		
160 		return set;
161 	}
