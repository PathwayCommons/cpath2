29  	// these are for 'search' queries
30  	private String queryString;
31  	private String type; // filter by
32  	private String[] organism; // filter by
33  	private String[] datasource; // filter by	
34  	private Integer page = 0; // search hits page #
35  	private boolean multi = false; //if true, and page>0 - all hits on pages 0..page; if true and page<0 - all hits (all pages).
36  	
37  	private static final int ALL_PAGES = -13; // a special value <0 for the page field

39  	protected MultiValueMap<String, String> getRequestParams() {
40  		MultiValueMap<String, String> request = new LinkedMultiValueMap<String, String>();
41  		if(queryString == null || queryString.isEmpty())
42  			throw new IllegalArgumentException("'queryString' parameter is required.");
43  		request.add(CmdArgs.q.name(), queryString);
44  		if(type != null)
45  			request.add(CmdArgs.type.name(), type);
46  		if(page != null && page > 0)
47  			request.add(CmdArgs.page.name(), page.toString());
48  		if(organism != null)
49  			request.put(CmdArgs.organism.name(), Arrays.asList(organism));
50  		if(datasource != null)
51  			request.put(CmdArgs.datasource.name(), Arrays.asList(datasource));
52  		return request;
53  	}

101 	/**
102 	 * The search results page number to retrieve, if there're more than one.
103 	 * (If a query was not quite specific, there are too many matches, 
104 	 * which are not returned all at once but pagination is used instead; 
105 	 * the greater page number, the less relevant hits are out there)
106 	 * 
107 	 * @param page number; the default is 0 (top hits)
108 	 * @return
109 	 * @throws IllegalArgumentException when page < 0
110 	 */
111 	public CPathSearchQuery page(int page) {
112 		if (page < 0)
113 			throw new IllegalArgumentException(
114 					"Negative page number");
115 
116 		this.page = page;
117 		this.multi = false;
118 
119 		return this;
120 	}

122 	/**
123 	 * Set to request all hits (all result pages).
124 	 * 
125 	 * @param lastPage
126 	 * @throws IllegalArgumentException when lastPage <= 0
127 	 * @return
128 	 */
129 	public CPathSearchQuery allPages() {
130 		multi = true;
131 		page = ALL_PAGES; //special val.
132 		if(queryString == null || queryString.isEmpty())
133 			queryString = "*";
134 		return this;
135 	}

137 	/**
138 	 * Set a number N > 0 to request
139 	 * top hits on pages 0..N.
140 	 * 
141 	 * @param lastPage
142 	 * @return
143 	 * @throws IllegalArgumentException when lastPage <= 0
144 	 */
145 	public CPathSearchQuery topPages(int lastPage) {
146 		if (page <= 0)
147 			throw new IllegalArgumentException(
148 				"The last page number must be greater than zero");
149 		multi = true;
150 		page = lastPage;
151 		
152 		return this;
153 	}

202 	@Override
203 	public SearchResponse result() throws CPathException {
204 		if(!multi) {
205 			return client.post(command, getRequestParams(), SearchResponse.class);
206 		} else {
207 			return multiPageResult();
208 		}
209 	}
210 	
211 	private SearchResponse multiPageResult() throws CPathException 
212 	{
213 	 	SearchResponse mulRes = null; //to return
214 		
215 		//when 'multi' is true, 'page'>0 means 
216 		//the last page to grab, and 'page'<0 - get all pages
217 	 	Integer last = this.page; 	 	
218 	 	//build and save the query parameters map (only once)
219 		MultiValueMap<String, String> request = getRequestParams();
220 		
221 		//btw: should never happen here due to how allPages, topPages(n), and page(n) are implemented
222 	 	assert multi==true && last!=0 : "bug: multiPageResult is called, whereas page==0, multi==true";
223 	 	// anyway, a shortcut (for the accidental bug/trivial case)
224 	 	if(last == 0) 
225 	 		return client.post(command, request, SearchResponse.class);
226 		
227 	 	Integer p = 0; //the first hits page # is always 0
228 	 	do {
229 			//update current 'page' # in the request map (but do not touch 'this.page' anymore!)
230 			request.put(CmdArgs.page.name(), Arrays.asList(p.toString())); //using 'add' instead of 'put' would be a bug
231 			SearchResponse res = client.post(command, request, SearchResponse.class);	 		
232 	 		if(res != null && !res.isEmpty()) { //collect hits
233 	 			if(mulRes == null) {
234 	 				mulRes = res;
235 	 				int totPages = res.numPages();
236 	 				//the last page # auto-correction
237 	 				if(last == ALL_PAGES || last >= totPages) 
238 	 					last = totPages-1;
239 	 			} else {
240 	 				mulRes.getSearchHit().addAll(res.getSearchHit());
241 	 			}	
242 	 		} else //should not happen (cpath2 returns error status when empty result)
243 	 			break; //there are no hits (no more)
244 	 		
245 	 		p++; //next page?
246 	 		
247 	 	} while(p <= last);
248 		
249 	 	if(mulRes != null)
250 	 		mulRes.setComment(mulRes.getComment() + 
251 	 			" This result combines top hits from multiple result pages: from 0 to " + last);
252 	 	
253 		return mulRes;
254 	}
